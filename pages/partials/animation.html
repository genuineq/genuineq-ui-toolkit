Animation

To create a CSS animation sequence, we have different sub-properties in the animation property in CSS : animation-delay, animation-direction, animation-duration, animation-iteration-count, animation-name, animation-play-state, animation-timing-function, animation-fill-mode.

HTML

<div class="width-1 height-2 animation-name-example animation-duration-4s 
animation-iteration-count-infinite"></div>

SCSS

$animation-name: ();
$animation-name: map-merge(
  (
    0: 0,
    "stretch": stretch,
    "example": example,
  ),
  $animation-name
);

$animation-duration: ();
$animation-duration: map-merge(
  (
    0: 0.3s,
    1: 1.5s,
    "4s": 4s,
  ),
  $animation-duration
);

$animation-timing-function: ();
$animation-timing-function: map-merge(
  (
    "ease": ease,
    "ease-in": ease-in,
    "ease-out": ease-out,
    "ease-in-out": ease-in-out,
    "linear": linear,
  ),
  $animation-timing-function
);

$animation-delay: ();
$animation-delay: map-merge(
  (
    0: 0,
  ),
  $animation-delay
);

$animation-direction: ();
$animation-direction: map-merge(
  (
    "alternate": alternate,
    "normal": normal,
  ),
  $animation-direction
);

$animation-iteration-count: ();
$animation-iteration-count: map-merge(
  (
    "infinite": infinite,
  ),
  $animation-iteration-count
);

$animation-fill-mode: ();
$animation-fill-mode: map-merge(
  (
    "none": none,
    "forwards": forwards,
    "backwards": backwards,
    "both": both,
  ),
  $animation-fill-mode
);

$animation-play-state: ();
$animation-play-state: map-merge(
  (
    "running": running,
    "paused": paused,
  ),
  $animation-play-state
);

Stroke

The stroke property in CSS is for adding a border to SVG shapes.

HTML

<svg class="fill-white" version="1.1" xmlns="http://www.w3.org/2000/svg">
  <circle class="stroke-danger stroke-width-2px stroke-linecap-butt stroke-dasharray-0" cx="60" cy="60" r="50"/>
</svg>
<svg class="fill-white" version="1.1" xmlns="http://www.w3.org/2000/svg">
  <circle class="stroke-info stroke-width-2px stroke-linecap-butt stroke-dasharray-0" cx="60" cy="60" r="50"/>
</svg>

SCSS

@each $attribute, $name in (stroke: stroke) {
  @each $key, $value in $colors {
    .#{$name}-#{$key} {
      #{$attribute}: $value;
    }
  }
}

$stroke-dasharray: ();
$stroke-dasharray: map-merge(
  (
    0: 0,
    2: 2,
    6: 6,
  ),
  $stroke-dasharray
);

$stroke-dashoffset: ();
$stroke-dashoffset: map-merge(
  (
    100: 100,
  ),
  $stroke-dashoffset
);

$stroke-linecap: ();
$stroke-linecap: map-merge(
  (
    "butt": butt,

);
